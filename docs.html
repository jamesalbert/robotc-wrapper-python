NAME

    robotc-wrapper - A RobotC wrapper written in Python, designed for the Vex PIC robot.

SYNOPSIS

    #!/usr/bin/env python

    from robotc import __init__, utils

    shortcut = utils.robotcwrapperutils()

    def create_a_lot_of_voids():
        shortcut.BASIC_MOVEMENTS()

    robot = __init__.robotcwrapper()

    def create_void():
        robot.start_void("function_name")
        robot.motor(1, 127)
        robot.motor(2, 127)
        robot.end()

    create_a_lot_of_voids()
    create_void()

LIST OF FUNCTIONS

    start_void( name )

        starts a void function in RobotC.

    start_task()

        starts the main task.

    start_if( condition )

        starts an if statement

    start_else_if( condition )

        starts an elseif statement

    if_sound()

        starts an if statement checking if sound is available

    if_active()

        starts an if statement checking if the Vex is active

    start_for( initial_count, end_count )

        starts a for loop

    start_while( condition )

        starts a while loop

    end()

        ends a block ( "}" )

    int_var( name, value )

        declares a integer variable

    char_var( name, value )

        declares a char variable

    long_var( name, value )

        declares a long variable

    battery( name )

        declares a variable, name, with the current battery level

    cos( radians ), sin( radians ), tan( radians )

        used to store value of cosine, sine, and tangent of radians respectively

    d_r( degrees ), r_d( radians )

        used to store value of degrees converted to radians and vise versa

    kill( task )

        declares the death of parameter, task

    mute()

        turns off sounds

    sound( frequency, duration )

        will play a tone with the given params

    tone( tone )

        param tone should be "buzz", "click", or "beep"

    sound_power( switch )

        a switch to turn sound on or off, switch should be true or false

    pragma( port, name, sensor )

        equivalent to "#pragma config(Sensor, in2, "button", sensorTouch);"
        should be delcared like "pragma(2, "button", "Touch")"

    reflect( port, switch )

        reflect a motor, port, if switch is true

    auto( switch )

        autonomous mode if switch is true

    inc_plus( var )

        returns var++

    inc_minus( var )

        returns var--

    motor( port, speed )

        sets up motor, port, to speed, speed

    speed_up( port, speed )

        returns motor, port, speed itself plus ( param ) speed

    clear_time()

        clears timer T1

    time_while( end_time )

        performs a while loop while time is less than or equal to end_time

    wait( duration )

        waits for duration ( in seconds )

    cont( port, channel )

        declares motor, port's, speed with channel, channel's, value

    call( function )

        calls a given function
